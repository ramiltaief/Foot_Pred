CREATE TABLE IF NOT EXISTS Foot
 (D DATE, home STRING, away STRING, 
FTHG INT, FTAG INT,FTR STRING,
HTHG INT,HTAG INT,HTR STRING,REF STRING,
HS INT,ASS INT,HST INT,AST INT,HF INT,AF INT,
HC INT,AC INT,HY INT,AY INT,HR INT,AR INT)
 ROW FORMAT DELIMITED
 FIELDS TERMINATED BY ','
 STORED AS TEXTFILE;

LOAD DATA LOCAL INPATH 'data.csv' OVERWRITE INTO TABLE Foot;
